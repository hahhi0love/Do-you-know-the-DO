<script>

    let audioCtx;

    let isLocked = false, isAllShow = false, isMinor = false, currentRoot = 0;

    const baseNotes = [{n:'C',k:'다',b:false,f:130.81},{n:'C#',k:'올림 다',b:true,f:138.59},{n:'D',k:'라',b:false,f:146.83},{n:'Eb',k:'내림 마',b:true,f:155.56},{n:'E',k:'마',b:false,f:164.81},{n:'F',k:'바',b:false,f:174.61},{n:'F#',k:'올림 바',b:true,f:185.00},{n:'G',k:'사',b:false,f:196.00},{n:'Ab',k:'내림 가',b:true,f:207.65},{n:'A',k:'가',b:false,f:220.00},{n:'Bb',k:'내림 나',b:true,f:233.08},{n:'B',k:'나',b:false,f:246.94}];

    const krNames = {"0M":"다장조","1M":"내림 라장조","2M":"라장조","3M":"내림 마장조","4M":"마장조","5M":"바장조","6M":"올림 바장조","7M":"사장조","8M":"내림 가장조","9M":"가장조","10M":"내림 나장조","11M":"나장조","0m":"가단조","2m":"나단조","5m":"라단조","7m":"마단조"};

    const signatures = {0:0,1:-5,2:2,3:-3,4:4,5:-1,6:6,7:1,8:-4,9:3,10:-2,11:5};

    const sigPos = {treble:{s:[-4,-1,-5,-2,1,-3,0],f:[0,-3,1,-2,-5,-1,-4]},bass:{s:[-2,1,-3,0,3,-1,2],f:[2,-1,3,0,-3,1,-2]},alto:{s:[-3,0,-4,-1,2,-2,1],f:[1,-2,2,-1,-4,0,-3]}};


    function initAudio() { if(!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)(); if(audioCtx.state==='suspended') audioCtx.resume(); }

    function playSound(freq) {

        initAudio(); const type = document.getElementById('inst-select').value;

        const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain();

        osc.frequency.value = freq;

        if(type==='piano') osc.type='triangle'; else if(type==='violin') osc.type='sawtooth'; else osc.type='sine';

        osc.connect(gain); gain.connect(audioCtx.destination);

        osc.start(); gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.8); osc.stop(audioCtx.currentTime + 1);

    }


    function drawStaff() {

        const svg = document.getElementById('staff-svg'); const clef = document.getElementById('clef-select').value;

        const rootIdx = currentRoot % 12; const sigVal = signatures[rootIdx];

        const steps = isMinor ? [0,2,3,5,7,8,10] : [0,2,4,5,7,9,11];

        const solNames = ['도','레','미','파','솔','라','시'];

        let html = '';

        for(let i=0; i<5; i++) html += `<line x1="20" y1="${60+i*10}" x2="580" y2="${60+i*10}" stroke="#cbd5e1" stroke-width="1.5"/>`;

        const cD = {treble:{t:'𝄞',y:105,s:65},bass:{t:'𝄢',y:90,s:55},alto:{t:'𝄡',y:100,s:60}}[clef];

        html += `<text x="30" y="${cD.y}" font-size="${cD.s}">${cD.t}</text>`;

        const p = sigPos[clef];

        if(sigVal>0) for(let i=0; i<sigVal; i++) html += `<text x="${95+i*14}" y="${85+p.s[i]*5}" font-size="24">♯</text>`;

        else if(sigVal<0) for(let i=0; i<Math.abs(sigVal); i++) html += `<text x="${95+i*14}" y="${85+p.f[i]*5}" font-size="24">♭</text>`;

        steps.forEach((step, i) => {

            const x = 230 + i*50; const y = 110 - i*5;

            html += `<circle cx="${x}" cy="${y}" r="6" fill="#1e293b" />`;

            if(y>=110) html += `<line x1="${x-12}" y1="110" x2="${x+12}" y2="110" stroke="#000" stroke-width="2"/>`;

            html += `<text x="${x}" y="145" text-anchor="middle" font-size="14" font-weight="bold" fill="${i===0?'#ef4444':'#475569'}">${solNames[i]}</text>`;

            html += `<text x="${x}" y="165" text-anchor="middle" font-size="12" fill="#94a3b8">${baseNotes[(rootIdx+step)%12].n}</text>`;

            html += `<text x="${x}" y="185" text-anchor="middle" font-size="12" fill="#64748b">${baseNotes[(rootIdx+step)%12].k}</text>`;

        });

        svg.innerHTML = html;

    }


    function updateUI(root) {

        if(!isLocked) currentRoot = root;

        const numKeys = parseInt(document.getElementById('key-count').value);

        const steps = isMinor ? [0,2,3,5,7,8,10] : [0,2,4,5,7,9,11];

        document.querySelectorAll('.solfege').forEach(el => el.innerText = '');

        document.querySelectorAll('.key').forEach(el => el.classList.remove('active'));

        for(let i=0; i<numKeys; i++) {

            const diff = ((i-currentRoot)%12+12)%12; const sIdx = steps.indexOf(diff);

            if(sIdx!==-1 && (isAllShow || (i>=currentRoot && i<currentRoot+12))) {

                document.getElementById(`s-${i}`).innerText = ['도','레','미','파','솔','라','시'][sIdx];

                document.querySelector(`[data-index="${i}"]`).classList.add('active');

            }

        }

        const keyKey = (currentRoot%12)+(isMinor?'m':'M');

        document.getElementById('key-name').innerText = (krNames[keyKey] || baseNotes[currentRoot%12].k + (isMinor?'단조':'장조'));

        drawStaff();

    }


    function initApp() {

        const numKeys = parseInt(document.getElementById('key-count').value);

        const kb = document.getElementById('keyboard'); kb.innerHTML = '';

        for(let i=0; i<numKeys; i++) {

            const note = baseNotes[i%12]; const key = document.createElement('div');

            key.className = `key ${note.b?'black':''}`; key.dataset.index = i;

            key.innerHTML = `<span class="solfege" id="s-${i}"></span>`;

            key.onmousedown = () => { initAudio(); playSound(note.f * Math.pow(2, Math.floor(i/12))); if(!isLocked) updateUI(i); };

            kb.appendChild(key);

        }

        updateUI(currentRoot);

    }

    function toggleMinor() { isMinor = !isMinor; document.getElementById('minor-btn').innerText = isMinor?"단조(Minor)":"장조(Major)"; document.getElementById('minor-btn').classList.toggle('minor-mode'); updateUI(currentRoot); }

    function toggleLock() { isLocked = !isLocked; document.getElementById('lock-btn').classList.toggle('active'); }

    function toggleAllShow() { isAllShow = !isAllShow; document.getElementById('all-show-btn').classList.toggle('active'); updateUI(currentRoot); }

    initApp();

</script>
